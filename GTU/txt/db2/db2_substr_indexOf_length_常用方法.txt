SELECT 
	SUBSTR(x111.TBL_NAME, 1, POSSTR(x111.TBL_NAME, '.') - 1) as TAB_SCHEMA, 
	SUBSTR(x111.TBL_NAME, POSSTR(x111.TBL_NAME, '.') + 1, LENGTH(x111.TBL_NAME)) as TAB_NAME, 
	X111.*,X110.DB_CONN
  FROM DBZU.DTZUX111 X111
  LEFT JOIN DBZU.DTZUX110 X110
  ON X111.MOD_NO = X110.MOD_NO
 WHERE 1=1
 [ and X111.MOD_NO = :MOD_NO ]
 [ AND X111.SER_NO = :SER_NO ]
 [ AND X111.SQL_NO = :SQL_NO ] 
 with ur

SELECT t.TBCREATOR AS schema,
       t.tbname AS table,
       (t.name),
       t.coltype,
       t.remarks,
       t.nulls,
       t.length
  FROM Sysibm.syscolumns t
 WHERE t.tbname = 'DTRDZ010' AND t.TBCREATOR = 'DBRD'
 order by 3
  fetch first 1000 rows only  ;

安裝npm
	
	sudo apt install npm


-----------------------------------------------------------------------------------------
建立專案

	cd /home/gtu001/桌面/ReactGtu
	mkdir src
	mkdir dist


-----------------------------------------------------------------------------------------
產生package.json
	
	cd /home/gtu001/桌面/ReactGtu
	npm init


-----------------------------------------------------------------------------------------
安裝webpack

	npm install webpack webpack-cli --save-dev


-----------------------------------------------------------------------------------------
寫webpack的設定檔

	touch webpack.config.js
		//內文
		const path = require('path');
		module.exports = {
		    entry: './index.js',
		    output: {
		        filename: 'index.bundle.js',
		        path: path.resolve(__dirname, './'),
		    }
		};

	touch index.js
		//內文
		console.log("Hello")

	編輯 package.json 
		  "scripts": {
		    "build": "webpack" <---新增
		  },
-----------------------------------------------------------------------------------------
重build

	npm run build

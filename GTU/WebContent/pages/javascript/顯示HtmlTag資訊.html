function hoverShowInfo(){ HTMLElement.prototype.after = function() { var argArr = Array.prototype.slice.call(arguments), docFrag = document.createDocumentFragment(); argArr.forEach(function (argItem) { var isNode = argItem instanceof Node; docFrag.appendChild(isNode ? argItem : document.createTextNode(String(argItem))); }); this.parentNode.insertBefore(docFrag, this.nextSibling); }; var hoverShow = function(ele){ var div1 = document.createElement("span"); div1.style.backgroundColor = "WHITE"; ele.after(div1); ele.addEventListener("mouseover", function(e){ var rect = ele.getBoundingClientRect(); div1.style.position = "absolute"; div1.style.left = (rect.left + rect.width); div1.style.top = (rect.top + rect.height); div1.style.visibility = "visible"; div1.style.display = "block"; var val = ""; if(ele.getAttribute("value")){ val = ", value = " + ele.getAttribute("value"); }else{ val = ", text = " + ele.innerText; } var text = ""; if(ele.getAttribute("name")){ text = "name = " + ele.getAttribute("name"); }else{ text = "id = " + ele.getAttribute("id"); } div1.innerHTML = text + val; }); ele.addEventListener("mouseout", function(e){ div1.style.visibility = "hidden"; div1.style.display = "none"; }); }; var arry = document.getElementsByTagName("input"); for(var ii = 0 ; ii < arry.length; ii ++){ hoverShow(arry[ii]); } arry = document.getElementsByTagName("select"); for(ii = 0 ; ii < arry.length; ii ++){ hoverShow(arry[ii]); } arry = document.getElementsByTagName("textarea"); for(ii = 0 ; ii < arry.length; ii ++){ hoverShow(arry[ii]); } }; $(document).ready(function(){ hoverShowInfo(); });

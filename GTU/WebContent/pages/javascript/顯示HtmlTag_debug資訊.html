function hoverShowInfo(){ HTMLElement.prototype.after = function() { var argArr = Array.prototype.slice.call(arguments), docFrag = document.createDocumentFragment(); argArr.forEach(function (argItem) { var isNode = argItem instanceof Node; docFrag.appendChild(isNode ? argItem : document.createTextNode(String(argItem))); }); this.parentNode.insertBefore(docFrag, this.nextSibling); }; var getMsgForSelect = function(ele){ try{ if(ele.tagName != "SELECT"){ return ele.options[ele.selectedIndex].value; } return null; }catch(e){ return null; } }; var getMsg = function(ele){ var msg = ""; if(ele.getAttribute("name")){ msg += "name = " + ele.getAttribute("name"); }else{ msg += "id = " + ele.getAttribute("id"); } if(ele.getAttribute("value")){ msg += "
value = " + ele.getAttribute("value"); }else if(getMsgForSelect(ele)){ msg += "
selectValue = " + getMsgForSelect(ele); }else{ msg += "
text = " + ele.innerText; } return msg; }; var hoverShow = function(ele){ var div1 = document.createElement("span"); div1.style.backgroundColor = "WHITE"; ele.after(div1); ele.addEventListener("mouseover", function(e){ var rect = ele.getBoundingClientRect(); div1.style.position = "absolute"; div1.style.left = (rect.left + rect.width); div1.style.top = (rect.top + rect.height); div1.style.visibility = "visible"; div1.style.display = "block"; div1.innerHTML = getMsg(ele); }); ele.addEventListener("mouseout", function(e){ div1.style.visibility = "hidden"; div1.style.display = "none"; }); }; var arry = document.getElementsByTagName("input"); for(var ii = 0 ; ii < arry.length; ii ++){ hoverShow(arry[ii]); } arry = document.getElementsByTagName("select"); for(ii = 0 ; ii < arry.length; ii ++){ hoverShow(arry[ii]); } arry = document.getElementsByTagName("textarea"); for(ii = 0 ; ii < arry.length; ii ++){ hoverShow(arry[ii]); } arry = document.getElementsByTagName("span"); for(ii = 0 ; ii < arry.length; ii ++){ var obj = arry[ii]; if(obj.getAttribute("id") || obj.getAttribute("name")){ hoverShow(obj); } } arry = document.getElementsByTagName("div"); for(ii = 0 ; ii < arry.length; ii ++){ var obj = arry[ii]; if(obj.getAttribute("id") || obj.getAttribute("name")){ hoverShow(obj); } } } $(document).ready(function(){ hoverShowInfo(); });
